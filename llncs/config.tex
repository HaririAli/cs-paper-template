\newcommand{\cmdKeywords}{
	\makeatletter
	    \def\lncsKeywords{}
	    \@for\varKeyword:={\varKeywords}\do{
	    	\ifx\lncsKeywords\empty
	    	\else
	    	\g@addto@macro\lncsKeywords{ \and}
	    	\fi
	    	\expandafter\g@addto@macro\expandafter\lncsKeywords\expandafter{\varKeyword}
    	}
	\makeatother
	\keywords{\lncsKeywords}
}
