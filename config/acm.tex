\def\insertAuthors{}

\makeatletter
\newcommand{\addAuthor}[2]{
	\g@addto@macro\insertAuthors{\author{#1}}
	\@for\varRef:={#2}\do{
		\expandafter\g@addto@macro\expandafter\insertAuthors\expandafter{\expandafter\getInst\expandafter{\varRef}}
	}
	\g@addto@macro\insertAuthors{ }
}

\newcommand{\addInst}[2]{\expandafter\gdef\csname my@inst@#1\endcsname{#2}}
\newcommand{\getInst}[1]{\csname my@inst@#1\endcsname}

\newcounter{instId}
\newcommand{\addInstitution}[4]{
	\stepcounter{instId}
	\addInst{\theinstId}{
		\affiliation{
			\institution{#1}
			\city{#2}
			\country{#3}
		}
		\email{#4}
	}
}
\makeatother

}
